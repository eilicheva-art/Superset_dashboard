-- добавление в HH дополнительных столбцов и удаление ненужных столбцов для построения дашборда
with t as(
select id, 
      technology,  
      name, 
      company,
      city, 
      salary_from, 
      salary_to,
      case 
        when salary_from = 0 then salary_to
        when salary_to = 0 then salary_from
        when salary_from <> 0 and salary_to <> 0 then (salary_from + salary_to)/2
      end salary_avg,
      case
        when salary_from +  salary_to = 0 then 'no'
        else 'yes'
      end salary_filled,
      salary_currency, 
      case 
        when salary_currency = 'RUR' then 1
        when salary_currency = 'USD' then 73.1385
        when salary_currency = 'EUR' then 86.2899
        when salary_currency = '' then 0
      end exchange_rate,
      experience, 
      employment,
      schedule, 
      published::date as published,
      now()::date - published::date as day_diff,
      case 
        when now()::date - published::date < 50 then 'менее 50 дней'
        when now()::date - published::date < 100 then 'от 50 до 100 дней'
        when now()::date - published::date >= 100 then '100 и более дней'
      end day_diff_group,
      key_skills, 
      description,
      row_number() over(partition by id order by published desc) as flag_for_drop_duplicates
from HH
)

select id,
      technology,
      name,
      company,
      city,
      salary_avg*exchange_rate as salary_avg_rur,
      salary_filled,
      experience,
      employment,
      schedule,
      published,
      day_diff_group
from t
where flag_for_drop_duplicates = 1


-- распарсим навыки
with t1 as(
select id, 
      technology,  
      key_skills,
      row_number() over(partition by id order by published desc) as flag_for_drop_duplicates
from HH
),

     t2 as(
select id,
       technology, 
       trim(lower(skill)) AS skills,
       count(id) over(partition by technology, trim(lower(skill))) as cnt_skill
from t1
array join splitByChar(',', key_skills) AS skill
where key_skills != '' and  flag_for_drop_duplicates=1
order by technology, lower(skill)
)

select technology,
       skills
from t2 
where cnt_skill >= 4
